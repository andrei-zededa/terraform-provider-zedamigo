#cloud-config

# Grow the root filesystem to maximum size, first the partition.
growpart:
  mode: auto
  devices: ['/']
  ignore_growroot_disabled: false

# Create an user with SSH authentication.
users:
  - name: ####USERNAME####
    sudo: ALL=(ALL) NOPASSWD:ALL
    shell: /bin/bash
    ssh_authorized_keys:
      - ####SSH_PUB_KEY####

# Set the password for the created user.
chpasswd:
  expire: false
  list: |
    ####USERNAME####:####PASSWORD####

# Remove all users except just the one creted and required system users.
userdel:
  remove: true
  force: true
  users:
    - ubuntu
    - cloud-user
    - ec2-user
    - centos
    - admin
    - default
    - user

# Keep only essential system users.
preserve_users:
  - root
  - systemd-network
  - systemd-resolve
  - systemd-timesync
  - messagebus
  - syslog
  - _apt
  - nobody
  - systemd-coredump
  - ####USERNAME####

# Set password authentication to false for enhanced security.
ssh_pwauth: false

# Install additional packages.
packages:
  - apt-transport-https
  - ca-certificates
  - curl
  - wget
  - gnupg
  - lsb-release
  - htop
  - vim
  - unzip
  - git
  - tmux
  - jq
  - net-tools
  - dnsutils
  - python3-scapy
  - iptables
  - stress-ng

# Update the package list and upgrade all packages on first boot.
package_update: true
package_upgrade: true

# Configure SSH(d).
ssh:
  install-server: true
  disable_root: true
  ssh_deletekeys: true
  ssh_genkeytypes: ['rsa', 'ecdsa', 'ed25519']

write_files:
  - path: /etc/hosts
    content: |
      127.0.0.1 localhost
      127.0.1.1 $zri.system.edge-instance.name
      ::1 localhost ip6-localhost ip6-loopback
      ff02::1 ip6-allnodes
      ff02::2 ip6-allrouters
    permissions: '0644'
    owner: root:root

  # Embed the netplan config instead of having a separate network-config.
  - path: /etc/netplan/01-network-config.yaml
    content: |
      network:
       version: 2
       ethernets:
         allens:
           match:
             name: en*
           dhcp4: true
           dhcp6: true
         alleths:
           match:
             name: eth*
           dhcp4: true
           dhcp6: true
    permissions: '0644'
    owner: root:root

  - path: /etc/sysctl.d/99-ip-forward.conf
    content: |
      net.ipv4.ip_forward = 1
    permissions: '0644'
    owner: root:root

hostname: $zri.system.edge-instance.name
fqdn: $zri.system.edge-instance.name

runcmd:
  # On Ubuntu there is a `/dev/root` symlink, but for example on Debian that
  # doesn't exist. Grow the filesystem after growing the partition.
  - 'resize2fs $(findmnt -n -o source /)'
  - mkdir -p /usr/local/bin
  - curl -sSLf --fail-early -o /root/msu-collect.tmp https://github.com/andrei-zededa/monitor-system-usage/releases/download/v0.0.4/msu-collect_0.0.4_linux_amd64 && echo "4fbee77afdd6cf75b40b51a438b2e0370221a69ae38b56f0c10501f10f27e2ac  /root/msu-collect.tmp" | sha256sum -c - && mv /root/msu-collect.tmp /usr/local/bin/msu-collect && chown root:root /usr/local/bin/msu-collect && chmod 0755 /usr/local/bin/msu-collect
  - rm /etc/netplan/50-cloud-init.yaml # Because we embedded the netplan config above instead of having a separate network-config.
  - netplan generate
  - netplan apply
  - sysctl -w net.ipv4.ip_forward=1
  - iptables -P FORWARD ACCEPT
  - iptables -A FORWARD -i eth1 -o eth2 -j ACCEPT
  - iptables -A FORWARD -i eth2 -o eth1 -j ACCEPT
  - iptables -A FORWARD -i enp0s3 -o enp0s4 -j ACCEPT
  - iptables -A FORWARD -i enp0s4 -o enp0s3 -j ACCEPT
  - iptables -A FORWARD -i enp0s4 -o enp0s5 -j ACCEPT
  - iptables -A FORWARD -i enp0s5 -o enp0s4 -j ACCEPT

# sudo ethtool -L enp0s4 combined 4
# sudo ethtool -L enp0s5 combined 4

# Final message on login screen
final_message: "System is ready after $UPTIME seconds."
